graph main {
    memory {
        counter : (4 + 2),
        result : ' ',
        flag : false
    }
    nodeDef {
        start :: startNode;
        mid;
        check;
        accept :: acceptNode;
        reject :: rejectNode;
    }
    transitions {
        start -> mid => conditions : (counter), operations : (counter);
        mid -> check => conditions : (flag), operations : (flag);
        check -> accept => conditions : (success), operations : (' ');
        check -> reject => conditions : (success), operations : (failure);
    }

}
