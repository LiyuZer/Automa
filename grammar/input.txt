graph main {
    memory {
        n: 5 < 7 < 3 < 2,
        num : 1
    }

    nodeDef {
        Start :: startNode;
        End :: acceptNode;
    }

    transitions {
        Start -> Start => conditions: (n > 0), operations: (num = num * n, n = n - 1, 'w', 5 * 7 * (9));
        Start -> End => conditions: (n == 0), operations: ();
    }

}
